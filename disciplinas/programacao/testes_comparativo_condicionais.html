if (<condição>) //caso verdadeiro
stateDiagram
	direction LR 
	[*] --> condição
	condição --> execução: verdadeiro
	condição --> [*]: falso
	execução --> [*]
        
if(<condição>){
	//caso verdadeiro
}
else {
	//caso falso
}
stateDiagram
	direction LR 
	[*] --> condição
	condição --> caso: verdadeiro
	condição --> senão: falso
	caso --> [*]
	senão: senão (else)
	caso: execução
        
if(<condição1>) {
	//caso verdadeiro
}
else if(<condição2>) {
	//caso verdadeiro 2
}
else if(<condição3>) {
	//caso verdadeiro 3
}
else {
	//se nenhuma das condições anteriores forem verdadeiras
}
stateDiagram
	direction LR 
	[*] --> condição1
	condição1 --> condição2: falso
	condição2 --> condição3: falso
	condição3 --> senão: falso
	condição1 --> execução1: verdadeiro
	condição2 --> execução2: verdadeiro
	condição3 --> execução3: verdadeiro
	execução1 --> [*]
	execução2 --> [*]
	execução3 --> [*]
	senão: senão (else)

O teste if em si não usa ;. A ordem de prioridade nos testes é respectiva.
